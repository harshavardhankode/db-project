DROP TABLE IF EXISTS Reviews;
DROP TABLE IF EXISTS Users;
DROP TABLE IF EXISTS Review;
DROP TABLE IF EXISTS Laptops;
DROP TABLE IF EXISTS Mobile_Phones;
DROP TABLE IF EXISTS Tablets;
DROP TABLE IF EXISTS Miscellaneous;
DROP TABLE IF EXISTS Electronic_Goods;
DROP TABLE IF EXISTS RAM;
DROP TABLE IF EXISTS processor;
DROP TABLE IF EXISTS graphics_card;

CREATE TABLE Users (Name VARCHAR(50), Username VARCHAR(50) PRIMARY KEY, Password VARCHAR(50), emailid VARCHAR(50) UNIQUE, Role VARCHAR(6) CHECK (Role = 'admin' OR Role = 'normal'));

CREATE TABLE Review (id SERIAL PRIMARY KEY, content VARCHAR(1000), rating INTEGER CHECK (rating >= 0 AND rating <= 10), time_stamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

CREATE TABLE RAM (id SERIAL PRIMARY KEY, size VARCHAR(100), speed VARCHAR(100), type VARCHAR(100));

CREATE TABLE processor (id SERIAL PRIMARY KEY, manf VARCHAR(100) NOT NULL, pname VARCHAR(100) NOT NULL, cores VARCHAR(100), clock_speed VARCHAR(100));

CREATE TABLE graphics_card (id SERIAL PRIMARY KEY, manf VARCHAR(100) NOT NULL, pname VARCHAR(100) NOT NULL, memory VARCHAR(100), type VARCHAR(100));

CREATE TABLE Electronic_Goods (PID SERIAL PRIMARY KEY, cost INTEGER CHECK (cost >= 0), weight REAL CHECK (weight >= 0), manf VARCHAR(50) NOT NULL, pname VARCHAR(50) NOT NULL, type CHAR(6) NOT NULL CHECK (type = 'laptop' OR type = 'mobile' OR type = 'tablet' OR type = 'miscel'));

CREATE TABLE Laptops (pid INTEGER DEFAULT 1 PRIMARY KEY REFERENCES Electronic_Goods(pid) ON DELETE CASCADE ON UPDATE CASCADE, cost INTEGER CHECK (cost >= 0), weight REAL CHECK (weight >= 0), manf VARCHAR(50) NOT NULL, pname VARCHAR(50) NOT NULL, os VARCHAR(100), speakers VARCHAR(100),network_adaptor VARCHAR(100), chipset VARCHAR(100), battery VARCHAR(100), screen_size VARCHAR(100), hard_disk VARCHAR(100), screen_resolution VARCHAR(100), ram_id INTEGER REFERENCES RAM(id) ON DELETE SET NULL ON UPDATE CASCADE, proc_id INTEGER REFERENCES processor(id) ON DELETE SET NULL ON UPDATE CASCADE, gcard_id INTEGER REFERENCES graphics_card(id) ON DELETE SET NULL ON UPDATE CASCADE);

CREATE TABLE Mobile_Phones (pid INTEGER DEFAULT 1 PRIMARY KEY REFERENCES Electronic_Goods(pid) ON DELETE CASCADE ON UPDATE CASCADE, cost INTEGER CHECK (cost >= 0), weight REAL CHECK (weight >= 0), manf VARCHAR(50) NOT NULL, pname VARCHAR(50) NOT NULL, os VARCHAR(100), internal_memory VARCHAR(100),screen_type VARCHAR(100), sim VARCHAR(100), screen_size VARCHAR(100), screen_resolution VARCHAR(100), ram_id INTEGER REFERENCES RAM(id) ON DELETE SET NULL ON UPDATE CASCADE, proc_id INTEGER REFERENCES processor(id) ON DELETE SET NULL ON UPDATE CASCADE);

CREATE TABLE Tablets (pid INTEGER DEFAULT 1 PRIMARY KEY REFERENCES Electronic_Goods(pid) ON DELETE CASCADE ON UPDATE CASCADE, cost INTEGER CHECK (cost >= 0), weight REAL CHECK (weight >= 0), manf VARCHAR(50) NOT NULL, pname VARCHAR(50) NOT NULL, os VARCHAR(100), internal_memory VARCHAR(100),screen_type VARCHAR(100), three_g VARCHAR(100), wifi VARCHAR(100), battery VARCHAR(100), screen_size VARCHAR(100), screen_resolution VARCHAR(100), ram_id INTEGER REFERENCES RAM(id) ON DELETE SET NULL ON UPDATE CASCADE, proc_id INTEGER REFERENCES processor(id) ON DELETE SET NULL ON UPDATE CASCADE, gcard_id INTEGER REFERENCES graphics_card(id) ON DELETE SET NULL ON UPDATE CASCADE);

CREATE TABLE Reviews (review_id INTEGER PRIMARY KEY REFERENCES review(id), egood_id INTEGER REFERENCES Electronic_Goods(pid), username VARCHAR(50) REFERENCES users(username)); 

CREATE TABLE Miscellaneous (pid INTEGER DEFAULT 1 PRIMARY KEY REFERENCES Electronic_Goods(pid), cost INTEGER CHECK (cost >= 0), weight REAL CHECK (weight >= 0), manf VARCHAR(50) NOT NULL, pname VARCHAR(50) NOT NULL, info VARCHAR(500));

CREATE OR REPLACE FUNCTION update_laptop_pid() RETURNS TRIGGER AS $_$	BEGIN INSERT into Electronic_goods(manf,pname,cost,weight,type) VALUES(New.manf,New.pname,New.cost,New.weight,'laptop');UPDATE Laptops SET pid = (SELECT e.pid FROM Electronic_Goods AS e WHERE New.manf = e.manf AND New.pname = e.pname) WHERE New.manf = Laptops.manf AND New.pname = Laptops.pname;		RETURN New;	END	$_$ LANGUAGE 'plpgsql';

CREATE TRIGGER LaptopTrig AFTER INSERT ON Laptops FOR EACH ROW EXECUTE PROCEDURE update_laptop_pid();

CREATE OR REPLACE FUNCTION update_tablet_pid() RETURNS TRIGGER AS $_$	BEGIN INSERT into Electronic_goods(manf,pname,cost,weight,type) VALUES(New.manf,New.pname,New.cost,New.weight,'tablet');UPDATE Tablets SET pid = (SELECT e.pid FROM Electronic_Goods AS e WHERE New.manf = e.manf AND New.pname = e.pname) WHERE New.manf = Tablets.manf AND New.pname = Tablets.pname;		RETURN New;	END	$_$ LANGUAGE 'plpgsql';
 	
CREATE TRIGGER TabletTrig AFTER INSERT ON Tablets FOR EACH ROW EXECUTE PROCEDURE update_tablet_pid();

CREATE OR REPLACE FUNCTION update_phone_pid() RETURNS TRIGGER AS $_$	BEGIN INSERT into Electronic_goods(manf,pname,cost,weight,type) VALUES(New.manf,New.pname,New.cost,New.weight,'mobile');UPDATE Mobile_Phones SET pid = (SELECT e.pid FROM Electronic_Goods AS e WHERE New.manf = e.manf AND New.pname = e.pname) WHERE New.manf = Mobile_Phones.manf AND New.pname = Mobile_Phones.pname;		RETURN New;	END	$_$ LANGUAGE 'plpgsql';
 	
CREATE TRIGGER MobileTrig AFTER INSERT ON Mobile_Phones FOR EACH ROW EXECUTE PROCEDURE update_phone_pid();

CREATE OR REPLACE FUNCTION update_misc_pid() RETURNS TRIGGER AS $_$	BEGIN INSERT into Electronic_goods(manf,pname,cost,weight,type) VALUES(New.manf,New.pname,New.cost,New.weight,'miscel');UPDATE Miscellaneous SET pid = (SELECT e.pid FROM Electronic_Goods AS e WHERE New.manf = e.manf AND New.pname = e.pname) WHERE New.manf = Miscellaneous.manf AND New.pname = Miscellaneous.pname;		RETURN New;	END	$_$ LANGUAGE 'plpgsql';
 	
CREATE TRIGGER MiscTrig AFTER INSERT ON Miscellaneous FOR EACH ROW EXECUTE PROCEDURE update_misc_pid();
